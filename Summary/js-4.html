for in : used to loop on countless property names
- Used with Objects
- can be used with Arrays and Strings but not recommended because 
  1- the order is not guaranteed
  2- it may include inherited properties
  3- indexes are strings not numbers

for ( let *key* in *object*) { 
*Code that will be executed on every key in the target*
object[key]
console.log(key) // prints the key
}

const fruits = ["apple", "panana", "orange"];

for (let i in fruits) {
  console.log(typeof (i), i, fruits[i]);
  innerHTML = ${i}$ = {fruits[i]};
  innerHTML += ${i}$ = {fruits[i]};
}
// Output:              | // Output
"string" "0" "apple"    | 2 orange
"string" "1" "panana"   | 0 apple
"string" "2" "orange"   | 1 panana
                        | 2 orange

For of : 
- Used to loop on iterable objects
- Used with : Arrays, Strings, Maps, Sets and...
- Don't use it with Objects

for (let *value* of *iterable*) {
*Code that will be executed on every elemnt in the target*
console.log(value) // prints the value
} 
//--------------------------------------------------
for each : 
1- built-in method that works only on arrays
2- executes a function on each element.
3- Always goes through all elements of the array
4- cann't use break or continue to control the loop

array.forEach(function(index, element, array) {
  console.log(index, element);
});
مشابه لـ continue

const arr = [1,2,3,4];

arr.forEach(n => {
  if (n === 2) return;
  console.log(n);
}); 
//Output: 1 3 4 

//--------------------------------------------------
break : Exits the loop immediately
//--------------------------------------------------
continue : Skips the current iteration and continues to the next one





Conditional ternary operator :

- same as if-else but shorter

- condition ? valueIfTrue : valueIfFalse

- can be nested

 const age = 18;

const canVote = age >= 18 ? "Yes" : "No";

console.log(canVote);
// Output: Yes
