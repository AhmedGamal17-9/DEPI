  attribute1: dataType;
  attribute2: dataType;

  constructor(parameter1: dataType, parameter2: dataType) {
    this.attribute1 = parameter1;
    this.attribute2 = parameter2;
  }

  method1(): returnType {
    // method logic
  }
}  



object => an instance of a class. It is a real entity created based on the class blueprint,
          with actual values assigned to its attributes and the ability to perform its methods.

let objectName = new className(parameter1Value, parameter2Value);


access modifiers
  1- public => can access the variable from anywhere
  2- private => can access the variable from inside the class only
  3- readonly => can read the variable from anywhere but cann't modify it from outside the class
  4- protected => can access the variable from inside the class and its subclasses only
